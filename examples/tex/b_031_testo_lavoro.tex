\begin{estrofa}

\numero{1} Axi com cell / qui'n lo somnis de lita

\numero{2} he son delit / de foll pensament ve

\numero{3} ne pren ami / quil temps passat, me te

\numero{4} limaginar / qu'altre be, noy habita

\numero{5} sentint venir / l'aguayt de, ma, dolor

\numero{6} sabent de cert / que'n ses mans, he de jaure

\numero{7} temps de venir / en ningun bem, pot caure

\numero{8} \c{c}o que's no res / en mi, es lo millor

\end{estrofa}



\begin{estrofa}

\numero{9} Al temps passat / me trobe, gran amor

\numero{10} amant no res / puix es tot, ja finit

\numero{11} d'aquest pensar / me sojorn, hem, delit

\numero{12} mas quant lo pert / sesfor\c{c}a, ma dolor

\numero{13} si com aquell / qui es jutjat, a, mort

\numero{14} he de lonch temps / la sab, es aconorta

\numero{15} he creurel fan / que li sera, estorta

\numero{16} el fan morir / sens vn punt, derecort

\end{estrofa}



\begin{estrofa}

\numero{17} Plagues, a deu / que mon pensar, fos mort

\numero{18} e que passas / ma vida, endurment

\numero{19} malament viu / qui te, son pensament

\numero{20} per enemich / fent li de'nuigs, report

\numero{21} he com lo vol / d'algun plaher, seruir

\numero{22} lin pren aixi / com dona'b son, infant

\numero{23} que, si veri li demana plorant

\numero{24} ha tan poch seny / que nol sab, contra dir

\end{estrofa}



\begin{estrofa}

\numero{25} Fora millor / ma dolor, sofferir

\numero{26} que no mesclar / poca part, de plaher

\numero{27} entr'aquells mals / quim giten, de saber

\numero{28} com del pensta\footnote{\textit{Per }pensat.} / plaher, me conue'xir

\numero{29} llas mon delit / dolor, se conuerteix

\numero{30} dobles l'affany / apres d'vn, poch repos

\numero{31} si col malalt / qui per, vn plahent, mos

\numero{32} tot son mejar / en dolor, se nodreix

\end{estrofa}



\begin{estrofa}

\numero{33} Com l'hermita / qui'nyorament, nol creix

\numero{34} d'aquells, amichs / que tenian, lo mon

\numero{35} tant ha lonch te\textit{m}ps / qu'n loch poblat, no fon

\numero{36} per fortuit / cas, vn dells l'apareix

\numero{37} qui los pasats / plahers, li renouella

\numero{38} si quel passat / present, li fa tornar

\numero{39} mas com sen part / lesfor\c{c}at, congoixar

\numero{40} lo be, com fuig / ab grans crits, mal, apella

\end{estrofa}


\begin{tornada}

\numero{42} absen\c{c}a es / lo verme qui, la guasta

\numero{43} si fer metat / durament, no contrasta

\numero{44} he creure poch / sil'enueios, consella
