\begin{estrofa}

\numero{1} A\textsc{x}i com cell / quen lo somnys delita

\numero{2} E son delit / de foll pensament ve

\numero{3} Ne pren a mi / quel temps passat me te

\numero{4} Limaginar / que altre be noy habita:

\numero{5} Sentint venir / laguayt de ma dolor

\numero{6} Sabent de cert / quen sens mans he de jaure

\numero{7} Temps per venir / en ningun bem pot caure

\numero{8} So ques no res / en mi es lo millor.

\end{estrofa}



\begin{estrofa}

\numero{9} \textparagraph\  Al temps passat / me trobe gran amor

\numero{10} Amant no res / puix tot es ja finit

\numero{11} Daquest pensar / me so jorn em delit

\numero{12} Mas quant lo pert / sesforsa ma dolor:

\numero{13} Si com aquell / qui es jutjat a mort

\numero{14} E de llonch temps / la sap e sa conorta

\numero{15} E creurel fan / que li sera estorta

\numero{16} El fan morir / sens vn punt de recort.

\end{estrofa}

\begin{estrofa}

\numero{17} \textparagraph\  Plagues a deu / que mon pensar fos mort

\numero{18} E que passas / ma vida en durment

\numero{19} Malament viu / qui te son pensament

\numero{20} Per enemich / fentli de mals ha port:

\numero{21} Que com lo vol / dalgun plaher seruir

\numero{22} Prenlin axi / com dona ab son infant

\numero{23} Que si veri / li demana plorant

\numero{24} Ha tant poch seny / que nol sab contradir.

\end{estrofa}



\begin{estrofa}

\numero{25} \textparagraph\  Fora millor / ma dolor soferir

\numero{26} Que no mesclar / poca part de plaher

\numero{27} Entre aquells mals / quem giten de saber

\numero{28} Com del plaher / passat me conue exir:

\numero{29} Llanc mon delit / dolor se conuerteix

\numero{30} Dobles lafany / apres dun poch repos

\numero{31} Si col malalt / que per vn plahent mos

\numero{32} Tot son menjar / en dolor se nodreix

\end{estrofa}
