\begin{estrofa}

\numero{9} \textparagraph\  Al temps passat / me trobe gran Amor

\numero{10} amant no res / puix es tot ja finit

\numero{11} de aquest pensar / me sojorn em'delit

\numero{12} mas com lo pert / se'sfor\c{c}a ma dolor.

\numero{13} Si com aquell / qui es jutjat a mort

\numero{14} e de llonch temps / la sab e se aconorta

\numero{15} e creure'l fan / que li sera estorta

\numero{16} el fan morir / sens vn punt de recort.

\end{estrofa}



\begin{estrofa}

\numero{18} e que passas / ma vida en durment

\numero{19} malament viu / qui te son pensament

\numero{20} per enemich / fent li de'nuigs report.
