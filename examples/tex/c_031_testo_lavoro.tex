\begin{estrofa}

\numero{1} Axi com cell / qu'n lo somni's delita

\numero{2} e son delit / de foll pensament ue

\numero{3} ne pren a mi / quil temps passat mete

\numero{4} l'immaginar / qu'altre be noy habita

\numero{5} sentint uenir / l'aguait de ma dolor

\numero{6} sabent de \c{c}ert / quen ses mans hede jaure

\numero{7} temps de uenir / en ningun be'm pot caure

\numero{8} \c{c}o qu'es no res / enmi es lo millor.

\end{estrofa}



\begin{estrofa}

\numero{9} Al temps passat / me trobe gran amor

\numero{10} amant no res / puix es tot ia finit

\numero{11} d'aquest pensar / me soiorn hem delita

\numero{12} mas quant lo pert / se sfor\c{c}a ma dolor

\numero{13} si com aquell / qui es iutiat a mort

\numero{14} e de lonch temps / la sab es aconorta

\numero{15} e creurel fan / queli sera estorta

\numero{16} el fan morir / sens un punt de recort.

\end{estrofa}

\begin{estrofa}

\numero{17} Plagues \`{a} deu / que mon pensar fos mort

\numero{18} e que passas / ma uida en durment

\numero{19} malament uiu / qui te son pensament

\numero{20} per enemich / fentli de nuigs report

\numero{21} e com lo uol / d'algun plaher seruir

\numero{22} lin pren axi / com don'ab son infant

\numero{23} que si ueri / li domana plorant

\numero{24} ha tan poch seny / que no sab contradir

\end{estrofa}



\begin{estrofa}

\numero{25} Fora millor / ma dolor sofferir

\numero{26} queno mesclar poca part de plaher

\numero{27} entre aquells mals / quem giten de psaber\footnote{Error per \textit{saber}.}

\numero{28} com del pensar / plaher me conue'xir

\numero{29} llas mon delit / dolor se conuerteix

\numero{30} doble's l'affany / apres d'un poch repos

\numero{31} si col malalt / qui per un plahent mos

\numero{32} rot son menjar / en dolor se nodreix

\end{estrofa}



\begin{estrofa}

\numero{34} d'aquells amichs / que tenia'n lo mon

\numero{35} tant ha lonch temps / qu'en loch pobat no fon

\numero{36} per fortuit cas / un d'ells l'appareix,

\numero{37} qui los passats / plahers li renouella

\numero{38} si quel passat / present li fa tornar

\numero{39} mas com sen part l'es forcat congoixar

\numero{40} lo be com fuig / ab gran crits mal apella

\end{estrofa}
