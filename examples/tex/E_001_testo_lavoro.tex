\begin{estrofa}

\numero{1} Axi com \c{c}ell / quin lo somnis delita

\numero{2} e son delit / de foll pensament ve

\numero{3} ne pren ami / quil temps passat mete

\numero{4} limaginar / qualtre be noy habita

\numero{5} Sentint venir / laguayt de ma dolor

\numero{6} sabent de cert / quen ses mans he dejaure

\numero{7} temps de venir / en ningun bem pot caure

\numero{8} \c{c}o ques no res / en mi es lo millor

\end{estrofa}



\begin{estrofa}

\numero{9} Al temps passat / me trobe gran amor

\numero{10} amant no res / puis es ja tot finit

\numero{11} daquest pensar / me soiorn hem delit

\numero{12} mas com lo pert / sesfor\c{c}a ma dolor

\numero{13} Si com aquell / qui es jutjat amort

\numero{14} e de llonch temps / la sap e sa conorta

\numero{15} e li es Dit / queli sera estorta

\numero{16} el fan morir / sens vn punt de recort

\end{estrofa}



\begin{estrofa}

\numero{17} Plagues a Deu / que mon pensar fos mort

\numero{18} e que passas / ma vida en durment

\numero{19} malament viu / qui te son pensament

\numero{20} per enemich / fent li denuigs aport

\numero{21} E com lo vol / dalgun plaher seruir

\numero{22} linpren axi / com donab son jnfant

\numero{23} que si veri / li demana plorant

\numero{24} ha tan poch seny / que nol sab contradir

\end{estrofa}

\begin{estrofa}

\numero{25} Fora millor / ma dolor sofferir

\numero{26} que no mesclar / poca part de plaher

\numero{27} entraquells mals / quim fan Deu car tener

\numero{28} com del plaer / passat me couexir

\numero{29} Las mon delit / dolor se conuerteix

\numero{30} dobles lafany / apres dun poch repos

\numero{31} si col malalt / qui per vn plazent mos

\numero{32} tot son menjar / endolor se nodreix

\end{estrofa}



\begin{estrofa}

\numero{33} Com lermita / quinyorament nol creix

\numero{34} daquells amichs / quauia en lo mon

\numero{35} tant ha llonch temps / quen lloch poblat no fon

\numero{36} fortuhit cas / vn dells li apareix

\numero{37} Qui los plahers / passats li renouella

\numero{38} si quel passat / present li fa tornar

\numero{39} mas com sen part / les for\c{c}at congoxar

\numero{40} lo be com fuig / ab gran crits mal apella

\end{estrofa}
